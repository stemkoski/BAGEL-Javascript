/**
 *  Entity is the super class for {@link Sprite} and {@link Group} objects.
 *  <br/>
 *  To be rendered in a {@link Game}, 
 *   entities must be added to the main Group of a {@link Screen},
 *   or a secondary Group that has added to the main Group.
 *  <br/>
 *  Entities store a reference to the Group they have been added to,
 *   to simplify removal from the Game (if needed).
 */
class Entity
{
	/**
	 * This object is abstract and is meant to be extended by other classes;
	 *  instances of this class should not be created.
	 * @constructor
	 */ 
	constructor()
	{
		this.parentGroup = null;
	}

	/**
	 * Get the Group that contains this entity.
	 * @return {Group} - the group containing this entity
	 */
	getGroup()
	{
		return this.parentGroup;
	}

	/**
	 * Draw this entity on the game canvas. 
	 * <br/>
	 * Must be implemented by extending class.
     * @param context - the graphics context object associated to the canvas where the entities will be drawn
	 */ 
	draw(context) 
	{ 
		throw new Error("draw method not implemented");
	}


}

export { Entity };
